 package com.increff.pos.dao;
 
 import com.increff.pos.pojo.OrdersPojo;
 import com.increff.pos.util.StringUtil;
 import org.springframework.stereotype.Repository;
 
 import javax.persistence.EntityManager;
 import javax.persistence.PersistenceContext;
 import javax.persistence.TypedQuery;
 import javax.transaction.Transactional;
 import java.time.LocalDate;
 import java.time.LocalDateTime;
 import java.util.List;
 
 @Repository
 public class OrderDao extends AbstractDao{
     private static String SELECT_ID = "select ordersPojo from OrdersPojo ordersPojo where orderId=:id";
     private static String SELECT_ALL = "select ordersPojo from OrdersPojo ordersPojo order by ordersPojo.orderId desc";
     private static String SELECT_ORDER_COUNT = "select count(op.orderId) from OrdersPojo op where op.orderTime = :date " +
                                 "and op.orderStatus = :orderStatus";
 
     @PersistenceContext
     private EntityManager entityManager;
 
     @Transactional
     public int insert(OrdersPojo ordersPojo) {
         entityManager.persist(ordersPojo);
         return ordersPojo.getOrderId();
     }
 
     public OrdersPojo select(int orderId) {
         TypedQuery<OrdersPojo> query = getQuery(SELECT_ID, OrdersPojo.class);
         query.setParameter("id", orderId);
         return getSingle(query);
     }
 
     public int selectCount(LocalDate date){
         LocalDateTime localDateTime= date.atStartOfDay();
         TypedQuery<Long> query = getQuery(SELECT_ORDER_COUNT , Long.class);
         query.setParameter("date", localDateTime);
         query.setParameter("orderStatus", StringUtil.invoiced);
         return getSingle(query).intValue();
     }
 
     public List<OrdersPojo> selectAll() {
         TypedQuery<OrdersPojo> query = getQuery(SELECT_ALL, OrdersPojo.class);
         return query.getResultList();
     }
 
 }
