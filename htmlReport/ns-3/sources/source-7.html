 package com.increff.pos.service;
 
 import com.increff.pos.dao.OrderDao;
 import com.increff.pos.pojo.OrdersPojo;
 import com.increff.pos.util.StringUtil;
 import org.springframework.beans.factory.annotation.Autowired;
 import org.springframework.stereotype.Service;
 
 import javax.transaction.Transactional;
 import java.time.LocalDateTime;
 import java.util.List;
 
 @Service
 public class OrderService {
     @Autowired
     private OrderDao orderDao;
 
     @Transactional
     public int add(){
         OrdersPojo ordersPojo = new OrdersPojo();
         ordersPojo.setOrderTime(LocalDateTime.now());
         ordersPojo.setOrderStatus(StringUtil.ongoing);
         return orderDao.insert(ordersPojo);
     }
 
     @Transactional(rollbackOn = ApiException.class)
     public OrdersPojo get(int orderId) throws ApiException {
         return getCheck(orderId);
     }
 
     @Transactional(rollbackOn  = ApiException.class)
     public void update(int orderId, String status) throws ApiException {
         OrdersPojo exOrdersPojo = getCheck(orderId);
         exOrdersPojo.setOrderStatus(status);
     }
 
     @Transactional
     public List<OrdersPojo> getAll() {
         return orderDao.selectAll();
     }
 
     @Transactional
     public void getCheckStatus(int orderId) throws ApiException {
         OrdersPojo ordersPojo = orderDao.select(orderId);
         if(ordersPojo.getOrderStatus().equals(StringUtil.invoiced)){
             throw new ApiException("Order is invoiced!");
         }
     }
 
     @Transactional
     public OrdersPojo getCheck(int orderId) throws ApiException {
         OrdersPojo ordersPojo = orderDao.select(orderId);
         if (ordersPojo == null) {
             throw new ApiException("Order with given ID does not exists!");
         }
         return ordersPojo;
     }
 
 }
