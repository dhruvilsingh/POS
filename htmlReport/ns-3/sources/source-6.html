 package com.increff.pos.service;
 
 import com.increff.pos.dao.InventoryDao;

 import com.increff.pos.pojo.OrderItemPojo;
 import com.increff.pos.util.StringUtil;
 import org.springframework.beans.factory.annotation.Autowired;
 import org.springframework.stereotype.Service;
 
 import javax.transaction.Transactional;
 import java.util.List;
 
 @Service
 public class OrderItemService {
 
     @Autowired
     private OrderItemDao orderItemDao;
 
     @Autowired
     private ProductDao productDao;
 
     @Autowired
     private InventoryDao inventoryDao;
 
     @Transactional(rollbackOn = ApiException.class)
     public int add(OrderItemPojo orderItemPojo, int productQuantity, double productMrp,
                     String productBarcode) throws ApiException {
         int updatedInventory = checkInput(orderItemPojo, productQuantity, productMrp, productBarcode);
         orderItemDao.insert(orderItemPojo);
         return updatedInventory;
     }
 
     @Transactional
     public List<OrderItemPojo> getAll(int orderId) {
         return orderItemDao.selectAll(orderId);
     }
 
     @Transactional
     public OrderItemPojo get(int itemId) throws ApiException {
         return getCheck(itemId);
     }
 
     @Transactional
     public void delete(int id) throws ApiException {
         getCheck(id);
         if(getAll(get(id).getOrderId()).size() == 1){
             throw new ApiException("Sorry! order cannot be empty");
         }
         orderItemDao.deleteId(id);
     }
 
     @Transactional(rollbackOn  = ApiException.class)
     public int update(int itemId, OrderItemPojo orderItemPojo, int productQuantity,
                        double productMrp) throws ApiException {
         OrderItemPojo exOrderItemPojo = getCheck(itemId);
         checkSellingPrice(orderItemPojo, productMrp);
         int updatedInventory = updateInventory(itemId, orderItemPojo, productQuantity);
         exOrderItemPojo.setSellingPrice(orderItemPojo.getSellingPrice());
         exOrderItemPojo.setProductQuantity(orderItemPojo.getProductQuantity());
         return updatedInventory;
     }
 
 
     @Transactional
     public OrderItemPojo getCheck(int itemNo) throws ApiException {
         OrderItemPojo orderItemPojo = orderItemDao.selectId(itemNo);
         if (orderItemPojo == null) {
             throw new ApiException("Item with given ID does not exist!");
         }
         return orderItemPojo;
     }
 
     private int updateInventory(int itemId , OrderItemPojo orderItemPojo,int productQuantity) throws ApiException {
         int oldQuantity = get(itemId).getProductQuantity();
         int newQuantity = orderItemPojo.getProductQuantity();
         if(oldQuantity == newQuantity){
             return oldQuantity;
         }
         int currentInventory = productQuantity;
         if(newQuantity < oldQuantity){
            return currentInventory + oldQuantity - newQuantity;
         }
         else if((newQuantity - oldQuantity) < currentInventory) {
             return currentInventory - newQuantity + oldQuantity;
         }
         else
             throw new ApiException("Sorry! Available up to: " + (oldQuantity+currentInventory));
     }
 
     private void checkSellingPrice(OrderItemPojo orderItemPojo, double productMrp) throws ApiException { //TODO : to remove this check.
         if (productMrp < orderItemPojo.getSellingPrice()) {
             throw new ApiException("Selling price cannot be more than MRP!!  (MRP = " + productMrp + ")");
         }
     }
 
     private int checkInput(OrderItemPojo orderItemPojo, int productQuantity, double productMrp, //TODO :  to check the inv validation in  inv.service.
                                  String productBarcode ) throws ApiException {
         if(orderItemPojo.getProductQuantity() > productQuantity){
             throw new ApiException(productQuantity + " items are left for " + productBarcode);
         }
         if(orderItemPojo.getSellingPrice() > productMrp){
             throw new ApiException("Selling price for " + productBarcode + " is greater than Mrp " + productMrp);
         }
         int updatedQuantity = productQuantity - orderItemPojo.getProductQuantity();
         return updatedQuantity;
     }
 
 }
